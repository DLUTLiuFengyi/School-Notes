安装依赖包
yum install -y conntrack ntpdate ntp ipvsadm ipset jq iptables curl sysstat

设置防火墙iptables并设置空规则
systemctl stop firewalld && systemctl disable firewalld
yum -y install iptables-services && systemctl start iptables && systemctl
enable iptables && iptables -F && service iptables save
关闭selinux
swapoff -a && sed -i '/ swap / s/^\(.*\)$/#\1/g' /etc/fstab
setenforce 0 && sed -i 's/^SELINUX=.*/SETLINUX=disabled/' /etc/selinux/config
